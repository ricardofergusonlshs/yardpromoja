-- Promote a signed-up user to content_uploader.
-- Replace uploader@example.com with the uploader's login email.

update public.profiles
set role = 'content_uploader'
where id = (
  select id
  from auth.users
  where email = 'uploader@example.com'
);

-- Verify result:
select id, email
from auth.users
where email = 'uploader@example.com';

select id, role, full_name, username
from public.profiles
where id = (
  select id
  from auth.users
  where email = 'uploader@example.com'
);
