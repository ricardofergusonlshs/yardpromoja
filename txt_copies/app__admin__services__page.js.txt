"use client";
import { useEffect, useMemo, useState } from "react";
import Link from "next/link";
import RequireRole from "@/app/components/RequireRole";
import { createClient } from "@/lib/supabaseClient";
const adminRoles = ["admin", "super_admin"];
const initialForm = { name: "", slug: "", description: "", amount_jmd: "", amount_usd: "", service_type: "general", active: true };
function makeSlug(value) { return String(value || "").toLowerCase().trim().replace(/[^a-z0-9]+/g, "-").replace(/^-+|-+$/g, ""); }
export default function AdminServicesPage() {
  const supabase = useMemo(() => createClient(), []); const [services, setServices] = useState([]); const [form, setForm] = useState(initialForm); const [editingId, setEditingId] = useState(""); const [loading, setLoading] = useState(true); const [message, setMessage] = useState("");
  async function loadServices() { setLoading(true); setMessage(""); try { const { data, error } = await supabase.from("paid_services").select("*").order("amount_jmd", { ascending: true }); if (error) throw error; setServices(data || []); } catch (error) { setMessage(error?.message || "Unable to load services."); } finally { setLoading(false); } }
  useEffect(() => { loadServices(); /* eslint-disable-next-line */ }, []);
  function updateField(field, value) { setForm((current) => ({ ...current, [field]: value, slug: field === "name" && !editingId ? makeSlug(value) : field === "slug" ? makeSlug(value) : current.slug })); }
  function editService(service) { setEditingId(service.id); setForm({ name: service.name || "", slug: service.slug || "", description: service.description || "", amount_jmd: String(service.amount_jmd || ""), amount_usd: String(service.amount_usd || ""), service_type: service.service_type || "general", active: Boolean(service.active) }); }
  function resetForm() { setEditingId(""); setForm(initialForm); }
  async function saveService(event) { event.preventDefault(); setMessage(""); if (!form.name.trim() || !form.slug.trim()) { setMessage("Name and slug are required."); return; } try { const payload = { name: form.name.trim(), slug: makeSlug(form.slug), description: form.description.trim(), amount_jmd: Number(form.amount_jmd || 0), amount_usd: form.amount_usd ? Number(form.amount_usd) : null, service_type: form.service_type.trim() || "general", active: Boolean(form.active), updated_at: new Date().toISOString() }; if (editingId) { const { error } = await supabase.from("paid_services").update(payload).eq("id", editingId); if (error) throw error; setMessage("Service updated."); } else { const { error } = await supabase.from("paid_services").insert([payload]); if (error) throw error; setMessage("Service created."); } resetForm(); await loadServices(); } catch (error) { setMessage(error?.message || "Unable to save service."); } }
  return <RequireRole allowedRoles={adminRoles}><main className="section"><div className="container"><div className="section-head"><div><p className="kicker">Admin Services</p><h1>Manage paid service pricing.</h1><p className="muted">Control the services customers can pay for on YardPromoJa.</p></div><Link className="btn btn-light" href="/admin/payments">Payments</Link></div>{message ? <div className="toast">{message}</div> : null}<form className="panel" onSubmit={saveService} style={{ display: "grid", gap: 12, marginBottom: 20 }}><h2>{editingId ? "Edit service" : "Create service"}</h2><div className="grid grid-3"><label>Name<input value={form.name} onChange={(event) => updateField("name", event.target.value)} /></label><label>Slug<input value={form.slug} onChange={(event) => updateField("slug", event.target.value)} /></label><label>Type<input value={form.service_type} onChange={(event) => updateField("service_type", event.target.value)} /></label></div><label>Description<textarea rows={3} value={form.description} onChange={(event) => updateField("description", event.target.value)} /></label><div className="grid grid-3"><label>Amount JMD<input type="number" value={form.amount_jmd} onChange={(event) => updateField("amount_jmd", event.target.value)} /></label><label>Amount USD<input type="number" value={form.amount_usd} onChange={(event) => updateField("amount_usd", event.target.value)} /></label><label>Active<select value={form.active ? "true" : "false"} onChange={(event) => updateField("active", event.target.value === "true")}><option value="true">Active</option><option value="false">Inactive</option></select></label></div><div className="dashboard-actions"><button className="btn btn-primary" type="submit">{editingId ? "Update Service" : "Create Service"}</button>{editingId ? <button className="btn btn-light" type="button" onClick={resetForm}>Cancel Edit</button> : null}</div></form>{loading ? <div className="toast">Loading services...</div> : null}<div className="table-card"><table><thead><tr><th>Service</th><th>Type</th><th>JMD</th><th>USD</th><th>Active</th><th>Action</th></tr></thead><tbody>{services.map((service) => <tr key={service.id}><td><strong>{service.name}</strong><br /><span className="muted">{service.slug}</span></td><td>{service.service_type}</td><td>${Number(service.amount_jmd || 0).toLocaleString()}</td><td>{service.amount_usd ? `$${Number(service.amount_usd).toLocaleString()}` : "—"}</td><td>{service.active ? "Yes" : "No"}</td><td><button className="btn btn-light" type="button" onClick={() => editService(service)}>Edit</button></td></tr>)}{!services.length ? <tr><td colSpan={6}><div className="empty"><h3>No services yet</h3><p className="muted">Create your first paid service above.</p></div></td></tr> : null}</tbody></table></div></div></main></RequireRole>;
}
