"use client";

import { useEffect, useMemo, useState } from "react";
import Link from "next/link";
import { createClient } from "@/lib/supabaseClient";

export default function PricingPage() {
  const supabase = useMemo(() => createClient(), []);
  const [services, setServices] = useState([]);
  const [loading, setLoading] = useState(true);
  const [message, setMessage] = useState("");

  useEffect(() => {
    let alive = true;
    async function loadServices() {
      setLoading(true);
      setMessage("");
      try {
        const { data, error } = await supabase
          .from("paid_services")
          .select("*")
          .eq("active", true)
          .order("amount_jmd", { ascending: true });
        if (error) throw error;
        if (alive) setServices(data || []);
      } catch (error) {
        if (alive) setMessage(error?.message || "Unable to load pricing.");
      } finally {
        if (alive) setLoading(false);
      }
    }
    loadServices();
    return () => { alive = false; };
  }, [supabase]);

  return (
    <main className="section">
      <div className="container">
        <div className="section-head">
          <div>
            <p className="kicker">Pricing</p>
            <h1>Promote smarter on YardPromoJa.</h1>
            <p className="muted">Choose a paid service for featured placement, premium visibility, weekend boost, sale offers, campaigns, and branded sharing.</p>
          </div>
          <Link className="btn btn-light" href="/services">Services</Link>
        </div>
        {message ? <div className="toast">{message}</div> : null}
        {loading ? <div className="toast">Loading pricing...</div> : null}
        {!loading && !services.length ? (
          <div className="empty"><h3>No paid services yet</h3><p className="muted">Admin can add services from /admin/services.</p></div>
        ) : null}
        <div className="grid" style={{ gridTemplateColumns: "repeat(auto-fit, minmax(240px, 1fr))", gap: 16 }}>
          {services.map((service) => (
            <article className="panel" key={service.id}>
              <p className="kicker">{service.service_type || "Service"}</p>
              <h2>{service.name}</h2>
              <p className="muted">{service.description}</p>
              <h3 style={{ marginTop: 14 }}>JMD ${Number(service.amount_jmd || 0).toLocaleString()}</h3>
              {service.amount_usd ? <p className="muted">Approx. USD ${Number(service.amount_usd).toLocaleString()}</p> : null}
              <Link className="btn btn-primary" href={`/checkout?service=${encodeURIComponent(service.slug)}`}>Pay Now</Link>
            </article>
          ))}
        </div>
      </div>
    </main>
  );
}
