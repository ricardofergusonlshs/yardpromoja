"use client";
import { useEffect, useMemo, useState } from "react";
import Link from "next/link";
import RequireRole from "@/app/components/RequireRole";
import { createClient } from "@/lib/supabaseClient";
import PaymentStatusBadge from "@/app/components/PaymentStatusBadge";
const adminRoles = ["admin", "super_admin"];
export default function AdminPaymentsPage() {
  const supabase = useMemo(() => createClient(), []);
  const [payments, setPayments] = useState([]); const [filter, setFilter] = useState("pending_review"); const [notes, setNotes] = useState({}); const [loading, setLoading] = useState(true); const [savingId, setSavingId] = useState(""); const [message, setMessage] = useState("");
  async function loadPayments() { setLoading(true); setMessage(""); try { let query = supabase.from("payments").select("*, service:paid_services(name,slug,service_type), user:profiles(full_name,username)").order("created_at", { ascending: false }); if (filter !== "all") query = query.eq("payment_status", filter); const { data, error } = await query; if (error) throw error; setPayments(data || []); } catch (error) { setMessage(error?.message || "Unable to load payments."); } finally { setLoading(false); } }
  useEffect(() => { loadPayments(); /* eslint-disable-next-line */ }, [filter]);
  function updateNote(id, value) { setNotes((current) => ({ ...current, [id]: value })); }
  async function updatePayment(payment, status) { setSavingId(payment.id); setMessage(""); try { const payload = { payment_status: status, admin_notes: notes[payment.id] || payment.admin_notes || "", updated_at: new Date().toISOString() }; if (status === "paid") payload.paid_at = new Date().toISOString(); const { error } = await supabase.from("payments").update(payload).eq("id", payment.id); if (error) throw error; await supabase.from("payment_events").insert([{ payment_id: payment.id, event_type: `admin_marked_${status}`, provider: payment.provider || payment.payment_method, payload }]); setMessage("Payment updated."); await loadPayments(); } catch (error) { setMessage(error?.message || "Unable to update payment."); } finally { setSavingId(""); } }
  return <RequireRole allowedRoles={adminRoles}><main className="section"><div className="container"><div className="section-head"><div><p className="kicker">Admin Payments</p><h1>Review customer payments.</h1><p className="muted">Approve bank transfers, reject invalid payments, and track paid services.</p></div><Link className="btn btn-light" href="/admin">Admin Dashboard</Link></div>{message ? <div className="toast">{message}</div> : null}<div className="dashboard-actions" style={{ marginBottom: 18 }}><select value={filter} onChange={(event) => setFilter(event.target.value)}><option value="all">All payments</option><option value="pending_review">pending review</option><option value="paid">paid</option><option value="rejected">rejected</option><option value="refunded">refunded</option><option value="cancelled">cancelled</option></select><Link className="btn btn-primary" href="/admin/services">Manage Services</Link></div>{loading ? <div className="toast">Loading payments...</div> : null}{!loading && !payments.length ? <div className="empty"><h3>No payments found</h3><p className="muted">Payments will appear here after customers submit them.</p></div> : null}<div style={{ display: "grid", gap: 16 }}>{payments.map((payment) => <section className="panel" key={payment.id}><div className="section-head"><div><p className="kicker">{payment.payment_method?.replaceAll("_", " ")} · {payment.currency || "JMD"}</p><h2>{payment.service?.name || "YardPromo service"}</h2><p className="muted">Amount: {payment.currency || "JMD"} ${Number(payment.amount || 0).toLocaleString()}</p></div><PaymentStatusBadge status={payment.payment_status} /></div><div className="promo-detail-list"><div><strong>Customer</strong><span>{payment.user?.full_name || payment.user?.username || payment.user_id}</span></div><div><strong>Bank Reference</strong><span>{payment.bank_reference || "No reference provided"}</span></div><div><strong>Proof</strong>{payment.proof_image_url ? <a href={payment.proof_image_url} target="_blank" rel="noreferrer">Open proof</a> : <span>No proof uploaded</span>}</div></div><label style={{ display: "block", marginTop: 14 }}>Admin notes<textarea rows={3} value={notes[payment.id] ?? payment.admin_notes ?? ""} onChange={(event) => updateNote(payment.id, event.target.value)} placeholder="Add approval/rejection notes..." /></label><div className="dashboard-actions" style={{ marginTop: 14 }}><button className="btn btn-primary" type="button" disabled={savingId === payment.id} onClick={() => updatePayment(payment, "paid")}>Mark Paid</button><button className="btn btn-light" type="button" disabled={savingId === payment.id} onClick={() => updatePayment(payment, "rejected")}>Reject</button><button className="btn btn-light" type="button" disabled={savingId === payment.id} onClick={() => updatePayment(payment, "refunded")}>Mark Refunded</button></div></section>)}</div></div></main></RequireRole>;
}
